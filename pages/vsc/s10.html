Command Quick Reference

คำสั่งทั้งหมดแบ่งตาม category — bookmark ไว้ใช้เวลา troubleshoot

VSC Management

// Create VSC with default PIN 12345678
tpmvscmgr.exe create /name "LabVSC" /pin default /adminkey random /generate

// Create VSC with custom PIN (minimum 8 chars)
tpmvscmgr.exe create /name "LabVSC" /pin prompt /adminkey random /generate

// Delete VSC by instance ID
tpmvscmgr.exe destroy /instance ROOT\SMARTCARDREADER\0000

Publishing & Importing

// Publish Root CA cert ลง Active Directory
certutil -dspublish -f "file.crt" RootCA

// Publish Root CA CRL ลง Active Directory
certutil -dspublish -f "file.crl" RootCA

// Import cert ลง Local Machine Trusted Root store
certutil -addstore -f root "file.crt"

// Import CRL ลง Local Machine Trusted Root store
certutil -addstore -f root "file.crl"

// Import CRL ลง Intermediate CA store
certutil -addstore -f CA "file.crl"

// Delete all cached CRL entries
certutil -urlcache crl delete

// Reset chain cache
certutil -setreg chain\ChainCacheResyncFiletime @now

Diagnostics

// Full VSC scan + chain trace (รัน CLIENT01, ป้อน PIN)
certutil -scinfo

// Show cached CRL entries — ถ้า empty แสดงว่า CRL ไม่เคย download
certutil -urlcache crl

// Verify certificate chain และ revocation status
certutil -verify -urlfetch cert.cer

// Force Issuing CA publish new CRL ทันที
certutil -crl

// Test CA RPC connectivity
certutil -ping

// Flush Kerberos ticket cache
klist purge
